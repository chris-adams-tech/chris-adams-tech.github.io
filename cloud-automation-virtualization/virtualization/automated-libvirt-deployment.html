import libvirt
import os
import time

# Define the base XML for the VM
base_xml = """
<domain type='kvm'>
  <name>{name}</name>
  <memory unit='KiB'>1048576</memory>
  <vcpu placement='static'>1</vcpu>
  <os>
    <type arch='x86_64' machine='pc-i440fx-2.9'>hvm</type>
    <boot dev='hd'/>
  </os>
  <devices>
    <disk type='file' device='disk'>
      <driver name='qemu' type='qcow2'/>
      <source file='/var/lib/libvirt/images/{name}.qcow2'/>
      <target dev='vda' bus='virtio'/>
    </disk>
    <interface type='network'>
      <source network='virbr3'/>
      <model type='virtio'/>
    </interface>
    <graphics type='spice' autoport='yes'/>
  </devices>
</domain>
"""

# Function to create a new VM
def create_vm(conn, name):
    xml = base_xml.format(name=name)
    conn.createXML(xml, 0)
    print(f"VM {name} created.")

# Function to install packages on the VM
def install_packages(vm_name):
    commands = [
        "apt-get update",
        "apt-get install -y spice-vdagent sysmon wazuh-agent"
    ]
    for cmd in commands:
        os.system(f"virsh -c qemu:///system qemu-agent-command {vm_name} '{cmd}'")
    print(f"Packages installed on {vm_name}.")

# Main function to deploy VMs
def deploy_vms(count):
    conn = libvirt.open('qemu:///system')
    for i in range(1, count + 1):
        vm_name = f"vm{i}"
        create_vm(conn, vm_name)
        time.sleep(10)  # Wait for the VM to be ready
        install_packages(vm_name)
    conn.close()

# Example usage
deploy_vms(5)

  </devices>

"""
  • This XML template defines the configuration for a new virtual machine (VM).
  • {name}: Placeholder for the VM name.
  • memory: Allocates 1 GB of memory to the VM.
  • vcpu: Allocates 1 virtual CPU to the VM.
  • disk: Specifies the disk image file and its type.
  • interface: Configures the network interface to use the virbr3 network.
  • graphics: Configures the VM to use SPICE for graphical output.

5. Main Function to Deploy VMs

def deploy_vms(count):
    conn = libvirt.open('qemu:///system')
    for i in range(1, count + 1):
        vm_name = f"vm{i}"
        create_vm(conn, vm_name)
        time.sleep(10)  # Wait for the VM to be ready
        install_packages(vm_name)
    conn.close()
  • deploy_vms: This function deploys the specified number of VMs.
  • libvirt.open: Connects to the local libvirt daemon.
  • for i in range(1, count + 1): Loops through the number of VMs to be created.
  • vm_name = f"vm{i}": Generates a unique name for each VM.
  • time.sleep(10): Waits for 10 seconds to ensure the VM is ready before installing packages.
  • conn.close(): Closes the connection to the libvirt daemon.

6. Example Usage

deploy_vms(5)
